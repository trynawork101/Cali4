// Generate icon as data URI using canvas
function makeIconDataURI(size) {
  const c = document.createElement('canvas');
  c.width = c.height = size;
  const ctx = c.getContext('2d');
  // bg
  ctx.fillStyle = '#0a0a0a';
  ctx.fillRect(0, 0, size, size);
  // rounded rect
  const pad = size * 0.12;
  const r = size * 0.18;
  ctx.fillStyle = '#c8f135';
  ctx.beginPath();
  ctx.moveTo(pad + r, pad);
  ctx.lineTo(size - pad - r, pad);
  ctx.quadraticCurveTo(size - pad, pad, size - pad, pad + r);
  ctx.lineTo(size - pad, size - pad - r);
  ctx.quadraticCurveTo(size - pad, size - pad, size - pad - r, size - pad);
  ctx.lineTo(pad + r, size - pad);
  ctx.quadraticCurveTo(pad, size - pad, pad, size - pad - r);
  ctx.lineTo(pad, pad + r);
  ctx.quadraticCurveTo(pad, pad, pad + r, pad);
  ctx.closePath();
  ctx.fill();
  // letter C
  ctx.fillStyle = '#0a0a0a';
  ctx.font = `bold ${size * 0.52}px Arial`;
  ctx.textAlign = 'center';
  ctx.textBaseline = 'middle';
  ctx.fillText('C', size / 2, size / 2);
  return c.toDataURL('image/png');
}

// ─── STATE ───────────────────────────────────────────────────────────────────
let state = JSON.parse(localStorage.getItem('cali_state') || '{}');
let activeDay = localStorage.getItem('cali_day') || 'mon';
let weekNum = parseInt(localStorage.getItem('cali_week') || '1');

function getKey(dayId, secId, exIdx) { return `${dayId}_${secId}_${exIdx}`; }

function getDayProgress(day) {
  if (day.rest) return { done: 0, total: 0 };
  let total = 0, done = 0;
  (day.sections || []).forEach(sec => {
    sec.exercises.forEach((_, i) => {
      total++;
      if (state[getKey(day.id, sec.id, i)]) done++;
    });
  });
  return { done, total };
}

function renderTabs() {
  const tabs = document.getElementById('dayTabs');
  tabs.innerHTML = '';
  DAYS.forEach(day => {
    const btn = document.createElement('button');
    const { done, total } = getDayProgress(day);
    const hasProgress = done > 0 && done < total;
    btn.className = `day-tab${day.rest ? ' rest-day' : ''}${day.id === activeDay ? ' active' : ''}${hasProgress ? ' has-progress' : ''}`;
    btn.innerHTML = `${day.label}<span class="dot"></span>`;
    btn.onclick = () => {
      activeDay = day.id;
      localStorage.setItem('cali_day', day.id);
      renderTabs();
      renderMain();
    };
    tabs.appendChild(btn);
  });
}

function renderMain() {
  const day = DAYS.find(d => d.id === activeDay);
  const main = document.getElementById('mainContent');
  document.getElementById('weekBadge').textContent = `WEEK ${weekNum}`;

  if (day.rest) {
    main.innerHTML = `
      <div class="day-header"><div>
        <div class="day-title">${day.title}</div>
        <div class="day-subtitle">${day.subtitle}</div>
      </div></div>
      <div class="rest-card">
        <div class="rest-icon">🛌</div>
        <div class="rest-title">Rest & Recover</div>
        <div class="rest-desc">${day.restNote}</div>
      </div>`;
    return;
  }

  const { done, total } = getDayProgress(day);
  const pct = total > 0 ? Math.round((done / total) * 100) : 0;
  const r = 26, circ = 2 * Math.PI * r;
  const offset = circ - (pct / 100) * circ;
  const allDone = done === total && total > 0;

  let html = `
    <div class="day-header">
      <div>
        <div class="day-title">${day.title}</div>
        <div class="day-subtitle">${day.subtitle}</div>
      </div>
      <div style="display:flex;align-items:center;gap:12px;">
        <button class="reset-btn" onclick="resetDay('${day.id}')">Reset</button>
        <div class="progress-ring-wrap">
          <div class="progress-ring">
            <svg width="64" height="64" viewBox="0 0 64 64">
              <circle class="track" cx="32" cy="32" r="${r}"/>
              <circle class="fill" cx="32" cy="32" r="${r}" stroke-dasharray="${circ}" stroke-dashoffset="${offset}"/>
            </svg>
            <div class="progress-pct">${pct}%</div>
          </div>
          <div class="progress-label">${done}/${total}</div>
        </div>
      </div>
    </div>`;

  if (allDone) {
    html += `<div class="complete-banner"><div class="complete-icon">🔥</div><div class="complete-text"><strong>Day Complete!</strong>${day.label} workout finished. Solid work.</div></div>`;
  }

  (day.sections || []).forEach(sec => {
    html += `<div class="section"><div class="section-label ${sec.type}-label">${sec.label}</div>`;
    sec.exercises.forEach((ex, i) => {
      const key = getKey(day.id, sec.id, i);
      const isDone = !!state[key];
      const tagHtml = ex.tag ? `<span class="ex-tag tag-${ex.tag}">${ex.tag === 'hs' ? '🤸 HS' : ex.tag === 'skill' ? 'skill' : 'core'}</span>` : '';
      html += `
        <div class="exercise-row${isDone ? ' done' : ''}" onclick="toggleEx('${day.id}','${sec.id}',${i})">
          <div class="check-box"></div>
          <div class="ex-info">
            <div class="ex-name">${ex.name}</div>
            <div class="ex-detail">${ex.detail}</div>
          </div>
          ${tagHtml}
        </div>`;
      if (ex.note) html += `<div class="note-row">${ex.note}</div>`;
    });
    html += `</div>`;
  });

  html += `<div class="deload-banner" style="margin-top:8px;"><div class="deload-icon">⚡</div><div class="deload-text">Every 4–6 weeks: cut all sets in half, keep handstand light, focus on perfect form.</div></div>`;
  main.innerHTML = html;
}

function toggleEx(dayId, secId, exIdx) {
  const key = getKey(dayId, secId, exIdx);
  state[key] = !state[key];
  localStorage.setItem('cali_state', JSON.stringify(state));
  renderTabs();
  renderMain();
}

function resetDay(dayId) {
  const day = DAYS.find(d => d.id === dayId);
  (day.sections || []).forEach(sec => {
    sec.exercises.forEach((_, i) => { delete state[getKey(dayId, sec.id, i)]; });
  });
  localStorage.setItem('cali_state', JSON.stringify(state));
  renderTabs();
  renderMain();
}

renderTabs();
renderMain();
